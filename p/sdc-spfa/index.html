#include <bits/stdc++.h>
#define endl "\n"
#define int long long
using namespace std;
const int N = 1e6+10;
const int inf = 1e18+10;

int n,m;
vector<pair<int,int>> g[N];
int dis[N],cnt[N],vis[N];
void solve()
{
    cin >> n >> m;
    for(int i=1;i<=m;i++)
    {
        int op;cin >> op;
        if(op==1)
        {
            int a,b,c;cin >> a >> b >> c;
            g[a].push_back({b,-c});
        }
        else if(op==2)
        {
            int a,b,c;cin >> a >> b >> c;
            g[b].push_back({a,c});
        }
        else if(op==3)
        {
            int a,b;cin >> a >> b;
            g[a].push_back({b,0});
            g[b].push_back({a,0});
        }
    }
    for(int i=1;i<=n;i++) 
    {
        dis[i] = inf;
        g[0].push_back({i,0});
    }
    queue<int> q;
    dis[0]=0;
    vis[0]=true;
    q.push(0);
    while(!q.empty())
    {
        int u = q.front();
        q.pop();
        vis[u]=false;
        for(auto [v,w] : g[u])
        {
            if(dis[v] > dis[u]+w)
            {
                dis[v] = dis[u]+w;
                cnt[v] = cnt[u]+1;
                if(cnt[v]>=n+1)
                {
                    cout << "No" << endl;
                    return;
                }
                if(!vis[v])
                {
                    q.push(v);
                    vis[v]=true;
                }
            }
        }
    }
    cout << "Yes" << endl;
}

signed main()
{
    ios::sync_with_stdio(0);
    cin.tie(0),cout.tie(0);
    int _ = 1;
    //cin >> _;
    while(_--)
    {
        solve();
    }
    return 0;
}
