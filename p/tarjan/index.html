int inStack[N];
int dfn[N],low[N];
stack<int> st;
vector<int> g[N];
int scc[N],sz[N];
int timestramp;
int cnt;
void dfs(int u)
{
    st.push(u);
    low[u] = dfn[u] = ++timestramp;
    inStack[u] = true;
    for(auto v : g[u])
    {
        if(!dfn[v])
        {
            dfs(v);
            low[u] = min(low[v],low[u]);
        }
        else if(inStack[v])
            low[u] = min(low[v],low[u]);
    }

    if(dfn[u]==low[u])
    {
        cnt++;
        do
        {
            inStack[st.top()] = false;
            scc[st.top()] = cnt;
            sz[cnt]++;
            st.pop();
        }while(st.top()!=u);
    }
}
