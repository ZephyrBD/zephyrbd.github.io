int n,m;
int a[N];
int b[N];
vector<int> arr[N];

int mergesort(int l,int r)
{
    if(l>=r) return 0;
    int mid = (l+r) >> 1;
    int res = mergesort(l,mid) + mergesort(mid+1,r);
    int i = l,j = mid+1,k = l;
    while(i<=mid&&j<=r)
    {
        if(a[i]<=a[j])
            b[k++] = a[i++];
        else
        {
            b[k++] = a[j++];
            res += mid - i + 1;
            //å¦‚æœå‘ç”Ÿä»jåé¢ä¸­æ‹¿å…ƒç´ ï¼Œè¯´æ˜aä¸­iä½ç½®å…¨æ¯”è¯¥å…ƒç´ å¤§
            //æ‰€ä»¥æ„æˆäº†mid-i+1ä¸ªé€†åºå¯¹
        }
    }
    while(i<=mid) b[k++] = a[i++];
    while(j<=r) b[k++] = a[j++];
    for(int i=l;i<=r;i++) a[i] = b[i];
    return res;
}

void solve()
{
    cin >> n >> m;
    for(int i=1;i<=n;i++)
    {
        cin>>a[i];
        arr[a[i]].push_back(i);
    }
    int res = mergesort(1,n);
    cout << res << endl;
    for(int k=1;k<=m-1;k++)
    {
        for(int j=0;j<(int)arr[m-k].size();j++)
        {
            res+=arr[m-k][j]-1-j;
            res-=(n-arr[m-k][j]-(arr[m-k].size()-j-1));
        }
        cout<<res<<endl;
    }
}

    {
        solve();
    }
