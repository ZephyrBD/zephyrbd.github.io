#include <bits/stdc++.h>
#define endl "\n"
#define int long long
using namespace std;
const int N = 1e5+10;

int n,m;
vector<int> g[N];
int color[N];
int tot1,tot2;
bool bfs(int x)
{
    queue<int> q;
    color[x] = 1;
    tot1 = 1;
    tot2 = 0;
    q.push(x);
    while(!q.empty())
    {
        int t = q.front();
        q.pop();

        for(auto v : g[t])
        {
            if(!color[v])
            {
                color[v] = color[t]%2+1;
                color[v] == 1 ? tot1++ : tot2++;
                q.push(v);
            }
            else if(color[v] == color[t]) return false;
        }
    }
    return true;
}

int ans;
void solve()
{
    cin >> n >> m;
    for(int i=1;i<=m;i++)
    {
        int u,v;cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u);
    }
    for(int i=1;i<=n;i++)
    {
        if(color[i]) continue;
        if(!bfs(i))
        {
            cout << "Impossible" << endl;
            return;
        }
        else ans += min(tot1,tot2);
    }
    cout << ans << endl;
}

signed main()
{
    ios::sync_with_stdio(0);
    cin.tie(0),cout.tie(0);
    int _ = 1;
    //cin >> _;
    while(_--)
    {
        solve();
    }
    return 0;
}
